    def matches?(response, &block)
      @response = response
      @document = Webrat::XML.document(response)
      @table_nodes = @document.xpath("//table")
      if (block_given?)
        # Asked: have_table { ... }
        block.bind(self).call
        return matches_conditions?
      else
        # Asked: have_table
        return @table_nodes.length > 0
      end
    end
